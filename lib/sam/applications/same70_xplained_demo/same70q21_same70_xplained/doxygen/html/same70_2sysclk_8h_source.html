324static inline uint32_t sysclk_get_main_hz(void)
325{
326#if (defined CONFIG_SYSCLK_DEFAULT_RETURNS_SLOW_OSC)
327 if (!sysclk_initialized ) {
328 return OSC_MAINCK_4M_RC_HZ;
329 }
330#endif
331
332 /* Config system clock setting */
333 if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_SLCK_RC) {
334 return OSC_SLCK_32K_RC_HZ;
335 } else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_SLCK_XTAL) {
336 return OSC_SLCK_32K_XTAL_HZ;
337 } else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_SLCK_BYPASS) {
338 return OSC_SLCK_32K_BYPASS_HZ;
339 } else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_MAINCK_4M_RC) {
340 return OSC_MAINCK_4M_RC_HZ;
341 } else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_MAINCK_8M_RC) {
342 return OSC_MAINCK_8M_RC_HZ;
343 } else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_MAINCK_12M_RC) {
344 return OSC_MAINCK_12M_RC_HZ;
345 } else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_MAINCK_XTAL) {
346 return OSC_MAINCK_XTAL_HZ;
347 } else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_MAINCK_BYPASS) {
348 return OSC_MAINCK_BYPASS_HZ;
349 }
350#ifdef CONFIG_PLL0_SOURCE
351 else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_PLLACK) {
352 return pll_get_default_rate(0);
353 }
354#endif
355
356#ifdef CONFIG_PLL1_SOURCE
357 else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_UPLLCK) {
358 return PLL_UPLL_HZ;
359 }
360#endif
361 else {
362 /* unhandled_case(CONFIG_SYSCLK_SOURCE); */
363 return 0;
364 }
365}

378static inline uint32_t sysclk_get_cpu_hz(void)
379{
380 /* CONFIG_SYSCLK_PRES is the register value for setting the expected */
381 /* prescaler, not an immediate value. */
382 return sysclk_get_main_hz() /
383 ((CONFIG_SYSCLK_PRES == SYSCLK_PRES_3) ? 3 :
384 (1 << (CONFIG_SYSCLK_PRES >> PMC_MCKR_PRES_Pos)));
385}

392static inline uint32_t sysclk_get_peripheral_hz(void)
