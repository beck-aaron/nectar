901 uint32_t ul_x2, uint32_t ul_y2)
902{
903 int dx, dy;
904 int i;
905 int xinc, yinc, cumul;
906 int x, y;
907
908 x = ul_x1;
909 y = ul_y1;
910 dx = ul_x2 - ul_x1;
911 dy = ul_y2 - ul_y1;
912 xinc = (dx > 0) ? 1 : -1;
913 yinc = (dy > 0) ? 1 : -1;
914 dx = abs(ul_x2 - ul_x1);
915 dy = abs(ul_y2 - ul_y1);
916
917 ili9488_draw_pixel(x, y);
918
919 if (dx > dy) {
920 cumul = dx >> 1;
921
922 for (i = 1; i <= dx; i++) {
923 x += xinc;
924 cumul += dy;
925
926 if (cumul >= dx) {
927 cumul -= dx;
928 y += yinc;
929 }
930
931 ili9488_draw_pixel(x, y);
932 }
933 } else {
934 cumul = dy >> 1;
935
936 for (i = 1; i <= dy; i++) {
937 y += yinc;
938 cumul += dx;
939
940 if (cumul >= dy) {
941 cumul -= dy;
942 x += xinc;
943 }
944
945 ili9488_draw_pixel(x, y);
946 }
947 }
