293sint8 hif_send(uint8 u8Gid,uint8 u8Opcode,uint8 *pu8CtrlBuf,uint16 u16CtrlBufSize,
294 uint8 *pu8DataBuf,uint16 u16DataSize, uint16 u16DataOffset)
295{
296 sint8 ret = M2M_ERR_SEND;
297 tstrHifHdr strHif;
298
299 strHif.u8Opcode = u8Opcode&(~NBIT7);
300 strHif.u8Gid = u8Gid;
301 strHif.u16Length = M2M_HIF_HDR_OFFSET;
302 if(pu8DataBuf != NULL)
303 {
304 strHif.u16Length += u16DataOffset + u16DataSize;
305 }
306 else
307 {
308 strHif.u16Length += u16CtrlBufSize;
309 }
310 if (strHif.u16Length <= M2M_HIF_MAX_PACKET_SIZE)
311 {
312 ret = hif_chip_wake();
313 if(ret == M2M_SUCCESS)
314 {
315 volatile uint32 reg, dma_addr = 0;
316 volatile uint16 cnt = 0;
317//#define OPTIMIZE_BUS
318/*please define in firmware also*/
319#ifndef OPTIMIZE_BUS
320 reg = 0UL;
321 reg |= (uint32)u8Gid;
322 reg |= ((uint32)u8Opcode<<8);
323 reg |= ((uint32)strHif.u16Length<<16);
324 ret = nm_write_reg(NMI_STATE_REG,reg);
325 if(M2M_SUCCESS != ret) goto ERR1;
326
327 reg = 0UL;
328 reg |= NBIT1;
329 ret = nm_write_reg(WIFI_HOST_RCV_CTRL_2, reg);
330 if(M2M_SUCCESS != ret) goto ERR1;
331#else
332 reg = 0UL;
333 reg |= NBIT1;
334 reg |= ((u8Opcode & NBIT7) ? (NBIT2):(0)); /*Data = 1 or config*/
335 reg |= (u8Gid == M2M_REQ_GROUP_IP) ? (NBIT3):(0); /*IP = 1 or non IP*/
336 reg |= ((uint32)strHif.u16Length << 4); /*length of pkt max = 4096*/
337 ret = nm_write_reg(WIFI_HOST_RCV_CTRL_2, reg);
338 if(M2M_SUCCESS != ret) goto ERR1;
339#endif
340 dma_addr = 0;
341
342 for(cnt = 0; cnt < 1000; cnt ++)
343 {
344 ret = nm_read_reg_with_ret(WIFI_HOST_RCV_CTRL_2,(uint32 *)&reg);
345 if(ret != M2M_SUCCESS) break;
346 /*
347 * If it takes too long to get a response, the slow down to
348 * avoid back-to-back register read operations.
349 */
350 if(cnt >= 500) {
351 if(cnt < 501) {
352 M2M_INFO("Slowing down...\n");
353 }
354 nm_bsp_sleep(1);
355 }
356 if (!(reg & NBIT1))
357 {
358 ret = nm_read_reg_with_ret(WIFI_HOST_RCV_CTRL_4,(uint32 *)&dma_addr);
359 if(ret != M2M_SUCCESS) {
360 /*in case of read error clear the DMA address and return error*/
361 dma_addr = 0;
362 goto ERR1;
363 }
364 /*in case of success break */
365 break;
366 }
367 }
368
369 if (dma_addr != 0)
370 {
371 volatile uint32 u32CurrAddr;
372 u32CurrAddr = dma_addr;
373 strHif.u16Length=NM_BSP_B_L_16(strHif.u16Length);
374 ret = nm_write_block(u32CurrAddr, (uint8*)&strHif, M2M_HIF_HDR_OFFSET);
375 if(M2M_SUCCESS != ret) goto ERR1;
376 u32CurrAddr += M2M_HIF_HDR_OFFSET;
377 if(pu8CtrlBuf != NULL)
378 {
379 ret = nm_write_block(u32CurrAddr, pu8CtrlBuf, u16CtrlBufSize);
380 if(M2M_SUCCESS != ret) goto ERR1;
381 u32CurrAddr += u16CtrlBufSize;
382 }
383 if(pu8DataBuf != NULL)
384 {
385 u32CurrAddr += (u16DataOffset - u16CtrlBufSize);
386 ret = nm_write_block(u32CurrAddr, pu8DataBuf, u16DataSize);
387 if(M2M_SUCCESS != ret) goto ERR1;
388 u32CurrAddr += u16DataSize;
389 }
390
391 reg = dma_addr << 2;
392 reg |= NBIT1;
393 ret = nm_write_reg(WIFI_HOST_RCV_CTRL_3, reg);
394 if(M2M_SUCCESS != ret) goto ERR1;
395 }
396 else
397 {
398 ret = hif_chip_sleep();
399 M2M_DBG("Failed to alloc rx size %d\r",ret);
400 ret = M2M_ERR_MEM_ALLOC;
401 goto ERR2;
402 }
403 }
404 else
405 {
406 M2M_ERR("(HIF)Failed to wakeup the chip\n");
407 goto ERR2;
408 }
409 }
410 else
411 {
412 M2M_ERR("HIF message length (%d) exceeds max length (%d)\n",strHif.u16Length, M2M_HIF_MAX_PACKET_SIZE);
413 ret = M2M_ERR_SEND;
414 goto ERR2;
415 }
416 /*actual sleep ret = M2M_SUCCESS*/
417 ret = hif_chip_sleep();
418 return ret;
419ERR1:
420 /*reset the count but no actual sleep as it already bus error*/
421 hif_chip_sleep_sc();
422ERR2:
423 /*logical error*/
424 return ret;
425}
434static sint8 hif_isr(void)
435{
436 sint8 ret = M2M_SUCCESS;
437 uint32 reg;
438 volatile tstrHifHdr strHif;
439
440 ret = nm_read_reg_with_ret(WIFI_HOST_RCV_CTRL_0, &reg);
441 if(M2M_SUCCESS == ret)
442 {
443 if(reg & 0x1) /* New interrupt has been received */
444 {
445 uint16 size;
446
447 /*Clearing RX interrupt*/
448 reg &= ~NBIT0;
449 ret = nm_write_reg(WIFI_HOST_RCV_CTRL_0,reg);
450 if(ret != M2M_SUCCESS)goto ERR1;
451 gstrHifCxt.u8HifRXDone = 1;
452 size = (uint16)((reg >> 2) & 0xfff);
453 if (size > 0) {
454 uint32 address = 0;
458 ret = nm_read_reg_with_ret(WIFI_HOST_RCV_CTRL_1, &address);
459 if(M2M_SUCCESS != ret)
460 {
461 M2M_ERR("(hif) WIFI_HOST_RCV_CTRL_1 bus fail\n");
462 goto ERR1;
463 }
464 gstrHifCxt.u32RxAddr = address;
465 gstrHifCxt.u32RxSize = size;
466 ret = nm_read_block(address, (uint8*)&strHif, sizeof(tstrHifHdr));
467 strHif.u16Length = NM_BSP_B_L_16(strHif.u16Length);
468 if(M2M_SUCCESS != ret)
469 {
470 M2M_ERR("(hif) address bus fail\n");
471 goto ERR1;
472 }
473 if(strHif.u16Length != size)
474 {
475 if((size - strHif.u16Length) > 4)
476 {
477 M2M_ERR("(hif) Corrupted packet Size = %u <L = %u, G = %u, OP = %02X>\n",
478 size, strHif.u16Length, strHif.u8Gid, strHif.u8Opcode);
479 ret = M2M_ERR_BUS_FAIL;
480 goto ERR1;
481 }
482 }
483
484 if(M2M_REQ_GROUP_WIFI == strHif.u8Gid)
485 {
486 if(gstrHifCxt.pfWifiCb)
487 gstrHifCxt.pfWifiCb(strHif.u8Opcode,strHif.u16Length - M2M_HIF_HDR_OFFSET, address + M2M_HIF_HDR_OFFSET);
488 else
489 M2M_ERR("WIFI callback is not registered\n");
490 }
491 else if(M2M_REQ_GROUP_IP == strHif.u8Gid)
492 {
493 if(gstrHifCxt.pfIpCb)
494 gstrHifCxt.pfIpCb(strHif.u8Opcode,strHif.u16Length - M2M_HIF_HDR_OFFSET, address + M2M_HIF_HDR_OFFSET);
495 else
496 M2M_ERR("Socket callback is not registered\n");
497 }
498 else if(M2M_REQ_GROUP_OTA == strHif.u8Gid)
499 {
500 if(gstrHifCxt.pfOtaCb)
501 gstrHifCxt.pfOtaCb(strHif.u8Opcode,strHif.u16Length - M2M_HIF_HDR_OFFSET, address + M2M_HIF_HDR_OFFSET);
502 else
503 M2M_ERR("Ota callback is not registered\n");
504 }
505 else if(M2M_REQ_GROUP_CRYPTO == strHif.u8Gid)
506 {
507 if(gstrHifCxt.pfCryptoCb)
508 gstrHifCxt.pfCryptoCb(strHif.u8Opcode,strHif.u16Length - M2M_HIF_HDR_OFFSET, address + M2M_HIF_HDR_OFFSET);
509 else
510 M2M_ERR("Crypto callback is not registered\n");
511 }
512 else if(M2M_REQ_GROUP_SIGMA == strHif.u8Gid)
513 {
514 if(gstrHifCxt.pfSigmaCb)
515 gstrHifCxt.pfSigmaCb(strHif.u8Opcode,strHif.u16Length - M2M_HIF_HDR_OFFSET, address + M2M_HIF_HDR_OFFSET);
516 else
517 M2M_ERR("Sigma callback is not registered\n");
518 }
519 else if(M2M_REQ_GROUP_SSL == strHif.u8Gid)
520 {
521 if(gstrHifCxt.pfSslCb)
522 gstrHifCxt.pfSslCb(strHif.u8Opcode,strHif.u16Length - M2M_HIF_HDR_OFFSET, address + M2M_HIF_HDR_OFFSET);
523 else
524 M2M_ERR("SSL callback is not registered\n");
525 }
526 else
527 {
528 M2M_ERR("(hif) invalid group ID\n");
529 ret = M2M_ERR_BUS_FAIL;
530 goto ERR1;
531 }
532 if(gstrHifCxt.u8HifRXDone)
533 {
534 M2M_ERR("(hif) host app didn't set RX Done <%u><%X>\n", strHif.u8Gid, strHif.u8Opcode);
535 ret = hif_set_rx_done();
536 if(ret != M2M_SUCCESS) goto ERR1;
537 }
538 }
539 else
540 {
541 M2M_ERR("(hif) Wrong Size\n");
542 ret = M2M_ERR_RCV;
543 goto ERR1;
544 }
545 }
546 else
547 {
548#ifndef WIN32
549 M2M_ERR("(hif) False interrupt %lx",reg);
550 goto ERR1;
551#else
552#endif
553 }
554 }
555 else
556 {
557 M2M_ERR("(hif) Failed to Read interrupt reg\n");
558 goto ERR1;
559 }
560
561ERR1:
562 return ret;
563}

581sint8 hif_handle_isr(void)
582{
583 sint8 ret = M2M_SUCCESS;
584
585 gstrHifCxt.u8Yield = 0;
586 while(gstrHifCxt.u8Interrupt && !gstrHifCxt.u8Yield)
587 {
588 /* Atomic decrement u8Interrupt since it takes multiple instructions to load, decrement and store,
589 * during which the ISR could fire again.
590 * If LEVEL interrupt is used instead of EDGE then the atomicity isn't needed since the interrupt
591 * is turned off in the ISR and back on again only after the interrupt has been serviced in hif_isr(). */
592
593#ifndef NM_LEVEL_INTERRUPT
594 nm_bsp_interrupt_ctrl(0);
595#endif
596
597 gstrHifCxt.u8Interrupt--;
598
599#ifndef NM_LEVEL_INTERRUPT
600 nm_bsp_interrupt_ctrl(1);
601#endif
602
603 uint8 retries = 5;
604 while(1)
605 {
606 ret = hif_isr();
607 if(ret == M2M_SUCCESS) {
608 /*we will try forever until we get that interrupt*/
609 /*Fail return errors here due to bus errors (reading expected values)*/
610 break;
611 } else {
612 retries--;
613 if(!retries)
614 {
615 M2M_ERR("(HIF) Failed to handle interrupt %d, aborting due to too many retries\n", ret);
616 break;
617 }
618 else
619 M2M_ERR("(HIF) Failed to handle interrupt %d try again... (%u)\n", ret, retries);
620 }
621 }
622 }
623
624 return ret;
625}
626/*
627* @fn hif_receive
628* @brief Host interface interrupt service routine
629* @param [in] u32Addr
630* Receive start address
631* @param [out] pu8Buf
632* Pointer to receive buffer. Allocated by the caller
633* @param [in] u16Sz
634* Receive buffer size
635* @param [in] isDone
636* If you don't need any more packets send True otherwise send false
637* @return The function shall return ZERO for successful operation and a negative value otherwise.
638*/
639sint8 hif_receive(uint32 u32Addr, uint8 *pu8Buf, uint16 u16Sz, uint8 isDone)
640{
641 sint8 ret = M2M_SUCCESS;
642 if((u32Addr == 0)||(pu8Buf == NULL) || (u16Sz == 0))
643 {
644 if(isDone)
645 {
646 /* set RX done */
647 ret = hif_set_rx_done();
648 }
649 else
650 {
651 ret = M2M_ERR_FAIL;
652 M2M_ERR(" hif_receive: Invalid argument\n");
653 }
654 goto ERR1;
655 }
656
657 if(u16Sz > gstrHifCxt.u32RxSize)
658 {
659 ret = M2M_ERR_FAIL;
660 M2M_ERR("APP Requested Size is larger than the received buffer size <%u><%lu>\n",u16Sz, gstrHifCxt.u32RxSize);
661 goto ERR1;
662 }
663 if((u32Addr < gstrHifCxt.u32RxAddr)||((u32Addr + u16Sz)>(gstrHifCxt.u32RxAddr + gstrHifCxt.u32RxSize)))
664 {
665 ret = M2M_ERR_FAIL;
666 M2M_ERR("APP Requested Address beyond the received buffer address and length\n");
667 goto ERR1;
668 }
669
670 /* Receive the payload */
671 ret = nm_read_block(u32Addr, pu8Buf, u16Sz);
672 if(ret != M2M_SUCCESS)goto ERR1;
673
674 /* check if this is the last packet */
675 if((((gstrHifCxt.u32RxAddr + gstrHifCxt.u32RxSize) - (u32Addr + u16Sz)) <= 0) || isDone)
676 {
677 /* set RX done */
678 ret = hif_set_rx_done();
679 }
680
681ERR1:
682 return ret;
683}
