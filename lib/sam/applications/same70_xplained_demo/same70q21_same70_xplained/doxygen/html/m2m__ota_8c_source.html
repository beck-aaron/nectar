82static void m2m_ota_cb(uint8 u8OpCode, uint16 u16DataSize, uint32 u32Addr)
83{
84 sint8 s8Ret = M2M_SUCCESS;
85 if(u8OpCode == M2M_OTA_RESP_NOTIF_UPDATE_INFO)
86 {
87 tstrOtaUpdateInfo strOtaUpdateInfo;
88 m2m_memset((uint8*)&strOtaUpdateInfo,0,sizeof(tstrOtaUpdateInfo));
89 s8Ret = hif_receive(u32Addr,(uint8*)&strOtaUpdateInfo,sizeof(tstrOtaUpdateInfo),0);
90 if(s8Ret == M2M_SUCCESS)
91 {
92 if(gpfOtaNotifCb)
93 gpfOtaNotifCb(&strOtaUpdateInfo);
94 }
95 }
96 else if (u8OpCode == M2M_OTA_RESP_UPDATE_STATUS)
97 {
98 tstrOtaUpdateStatusResp strOtaUpdateStatusResp;
99 m2m_memset((uint8*)&strOtaUpdateStatusResp,0,sizeof(tstrOtaUpdateStatusResp));
100 s8Ret = hif_receive(u32Addr, (uint8*) &strOtaUpdateStatusResp,sizeof(tstrOtaUpdateStatusResp), 0);
101 if(s8Ret == M2M_SUCCESS)
102 {
103 if(gpfOtaUpdateCb)
104 gpfOtaUpdateCb(strOtaUpdateStatusResp.u8OtaUpdateStatusType,strOtaUpdateStatusResp.u8OtaUpdateStatus);
105 }
106 }
107 else if (u8OpCode == M2M_OTA_RESP_HOST_FILE_STATUS)
108 {
109 tstrOtaHostFileGetStatusResp strOtaHostFileGetStatusResp = {0};
110 s8Ret = hif_receive(u32Addr, (uint8*)&strOtaHostFileGetStatusResp, sizeof(tstrOtaHostFileGetStatusResp), 1);
111 if(M2M_SUCCESS == s8Ret)
112 {
113 if(strOtaHostFileGetStatusResp.u8OtaFileGetStatus == OTA_STATUS_SUCCESS) {
114 gu8CurrFileHandlerID = strOtaHostFileGetStatusResp.u8CFHandler;
115 }
116 }
117 }
118 else if (u8OpCode == M2M_OTA_RESP_HOST_FILE_DOWNLOAD)
119 {
120 tstrOtaHostFileGetStatusResp strOtaHostFileGetStatusResp = {0};
121 s8Ret = hif_receive(u32Addr, (uint8*)&strOtaHostFileGetStatusResp, sizeof(tstrOtaHostFileGetStatusResp), 1);
122 if(M2M_SUCCESS == s8Ret)
123 {
124 if(strOtaHostFileGetStatusResp.u8OtaFileGetStatus == OTA_STATUS_SUCCESS) {
125 gu8CurrFileHandlerID = strOtaHostFileGetStatusResp.u8CFHandler;
126 M2M_INFO("Generated HostFileHandlerID is %u\n", gu8CurrFileHandlerID);
127 }
128
129 if(gpfHFDGetCb) {
130 gpfHFDGetCb(strOtaHostFileGetStatusResp.u8OtaFileGetStatus, gu8CurrFileHandlerID, strOtaHostFileGetStatusResp.u32OtaFileSize);
131 gpfHFDGetCb = NULL;
132 }
133 }
134 }
135 else if (u8OpCode == M2M_OTA_RESP_HOST_FILE_READ)
136 {
137 tstrOtaHostFileReadStatusResp strOtaHostFileReadStatusResp;
138 m2m_memset((uint8*)&strOtaHostFileReadStatusResp, 0, sizeof(tstrOtaHostFileReadStatusResp));
139 s8Ret = hif_receive(u32Addr, (uint8*)&strOtaHostFileReadStatusResp, sizeof(tstrOtaHostFileReadStatusResp), 1);
140 if(M2M_SUCCESS == s8Ret)
141 if(gpfHFDReadCb)
142 gpfHFDReadCb(strOtaHostFileReadStatusResp.u8OtaFileReadStatus, strOtaHostFileReadStatusResp.pFileBuf, strOtaHostFileReadStatusResp.u16FileBlockSz);
143 }
144 else if(u8OpCode == M2M_OTA_RESP_HOST_FILE_ERASE)
145 {
146 tstrOtaHostFileEraseStatusResp strOtaHostFileEraseStatusResp = {0};
147 s8Ret = hif_receive(u32Addr, (uint8*)&strOtaHostFileEraseStatusResp, sizeof(tstrOtaHostFileEraseStatusResp), 1);
148 if(M2M_SUCCESS == s8Ret)
149 {
150 if(gpfHFDEraseCb)
151 {
152 gpfHFDEraseCb(strOtaHostFileEraseStatusResp.u8OtaFileEraseStatus);
153 gpfHFDEraseCb = NULL;
154 }
155 }
156 }
157 else
158 {
159 M2M_ERR("Invalid OTA resp %d ?\n",u8OpCode);
160 }
161}

449NMI_API sint8 m2m_ota_host_file_read_spi(uint8 u8Handler, uint8 *pu8Buff, uint32 u32Offset, uint32 u32Size)
450{
451 static uint32 u32FlashHFDStart = 0;
452 static uint32 u32FlashHFDSize = 0;
453 sint8 s8Ret = M2M_ERR_INVALID_ARG;
454 if((u8Handler == HFD_INVALID_HANDLER) || (NULL == pu8Buff)) goto EXIT;
455
456 if(WIFI_STATE_INIT != m2m_wifi_get_state())
457 {
458 s8Ret = M2M_ERR_FAIL;
459 M2M_ERR("WINC is not in an appropriate state for this operation!\n");
460 goto EXIT;
461 }
462
463 if((u32FlashHFDStart == 0) || (u32FlashHFDSize == 0))
464 {
465 s8Ret = spi_flexible_flash_find_section(ENTRY_ID_HOSTFILE, &u32FlashHFDStart, &u32FlashHFDSize);
466 if(M2M_SUCCESS != s8Ret) goto EXIT;
467 }
468
469 s8Ret = spi_flash_read(pu8Buff, u32FlashHFDStart, 4);
470
471 if((M2M_SUCCESS != s8Ret) || (pu8Buff[0] != u8Handler)) goto EXIT;
472
473 if((u32Offset >= u32FlashHFDSize) ||
474 (u32Size > u32FlashHFDSize) ||
475 ((u32Offset + u32Size) >= u32FlashHFDSize))
476 {
477 s8Ret = M2M_ERR_FAIL;
478 goto EXIT;
479 }
480
481 s8Ret = spi_flash_read(pu8Buff, u32FlashHFDStart + FLASH_SECTOR_SZ + u32Offset, u32Size);
482
483 if(M2M_SUCCESS != s8Ret)
484 M2M_ERR("Unable to read SPI Flash\n");
485
486EXIT:
487 return s8Ret;
488}
