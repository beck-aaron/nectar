227uint32_t rtt_write_alarm_time(Rtt *p_rtt, uint32_t ul_alarm_time)
228{
229 uint32_t flag;
230
231 flag = p_rtt->RTT_MR & RTT_MR_ALMIEN;
232
233 rtt_disable_interrupt(RTT, RTT_MR_ALMIEN);
234
239 if(ul_alarm_time == 0) {
240 p_rtt->RTT_AR = 0xFFFFFFFF;
241 }
242 else {
243 p_rtt->RTT_AR = ul_alarm_time - 1;
244 }
245
246 if (flag) {
247 rtt_enable_interrupt(RTT, RTT_MR_ALMIEN);
248 }
249
250 return 0;
251}
