135void nm_bsp_register_isr(tpfNmBspIsr pfIsr)

137 gpfIsr = pfIsr;

139 /* Configure PGIO pin for interrupt from SPI slave, used when slave has data to send. */
140 pmc_enable_periph_clk(CONF_WINC_SPI_INT_PIO_ID);
141 pio_configure_pin(CONF_WINC_SPI_INT_PIN, PIO_TYPE_PIO_INPUT);
142 pio_pull_up(CONF_WINC_SPI_INT_PIO, CONF_WINC_SPI_INT_MASK, PIO_PULLUP);
143 /*Interrupt on falling edge*/
144 pio_handler_set(CONF_WINC_SPI_INT_PIO, CONF_WINC_SPI_INT_PIO_ID,
145 CONF_WINC_SPI_INT_MASK, PIO_PULLUP | PIO_IT_FALL_EDGE, chip_isr);
146 pio_get_interrupt_status(CONF_WINC_SPI_INT_PIO);
147 pio_enable_interrupt(CONF_WINC_SPI_INT_PIO, CONF_WINC_SPI_INT_MASK);
148 NVIC_EnableIRQ((IRQn_Type) CONF_WINC_SPI_INT_PIO_ID);
149 pio_handler_set_priority(CONF_WINC_SPI_INT_PIO, (IRQn_Type)CONF_WINC_SPI_INT_PIO_ID,
150 CONF_WINC_SPI_INT_PRIORITY);
151}
