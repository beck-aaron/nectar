109static inline void osc_enable(uint32_t ul_id)
110{
111 switch (ul_id) {
112 case OSC_SLCK_32K_RC:
113 break;
114
115 case OSC_SLCK_32K_XTAL:
116 pmc_switch_sclk_to_32kxtal(PMC_OSC_XTAL);
117 break;
118
119 case OSC_SLCK_32K_BYPASS:
120 pmc_switch_sclk_to_32kxtal(PMC_OSC_BYPASS);
121 break;
122
123
124 case OSC_MAINCK_4M_RC:
125 pmc_switch_mainck_to_fastrc(CKGR_MOR_MOSCRCF_4_MHz);
126 break;
127
128 case OSC_MAINCK_8M_RC:
129 pmc_switch_mainck_to_fastrc(CKGR_MOR_MOSCRCF_8_MHz);
130 break;
131
132 case OSC_MAINCK_12M_RC:
133 pmc_switch_mainck_to_fastrc(CKGR_MOR_MOSCRCF_12_MHz);
134 break;
135
136
137 case OSC_MAINCK_XTAL:
138 pmc_switch_mainck_to_xtal(PMC_OSC_XTAL,
139 pmc_us_to_moscxtst(BOARD_OSC_STARTUP_US,
140 OSC_SLCK_32K_RC_HZ));
141 break;
142
143 case OSC_MAINCK_BYPASS:
144 pmc_switch_mainck_to_xtal(PMC_OSC_BYPASS,
145 pmc_us_to_moscxtst(BOARD_OSC_STARTUP_US,
146 OSC_SLCK_32K_RC_HZ));
147 break;
148 }
149}
