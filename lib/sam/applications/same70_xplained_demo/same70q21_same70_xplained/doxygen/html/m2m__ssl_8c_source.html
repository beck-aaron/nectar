70static void m2m_ssl_cb(uint8 u8OpCode, uint16 u16DataSize, uint32 u32Addr)
71{
72 sint8 s8tmp = M2M_SUCCESS;
73 switch(u8OpCode)
74 {
75 case M2M_SSL_REQ_ECC:
76 {
77 tstrEccReqInfo strEccREQ;
78 s8tmp = hif_receive(u32Addr, (uint8 *)&strEccREQ, sizeof(tstrEccReqInfo), 0);
79 if(s8tmp == M2M_SUCCESS)
80 {
81 if(gpfAppSSLCb)
82 {
83 gu32HIFAddr = u32Addr + sizeof(tstrEccReqInfo);
84 gpfAppSSLCb(M2M_SSL_REQ_ECC, &strEccREQ);
85 }
86 }
87 }
88 break;
89 case M2M_SSL_RESP_SET_CS_LIST:
90 {
91 tstrSslSetActiveCsList strCsList;
92 s8tmp = hif_receive(u32Addr, (uint8 *)&strCsList, sizeof(tstrSslSetActiveCsList), 0);
93 if(s8tmp == M2M_SUCCESS)
94 {
95 if(gpfAppSSLCb)
96 gpfAppSSLCb(M2M_SSL_RESP_SET_CS_LIST, &strCsList);
97 }
98 }
99 break;
100 case M2M_SSL_RESP_WRITE_OWN_CERTS:
101 {
102 tstrTlsSrvChunkHdr strTlsSrvChunkRsp;
103 uint8 bCallApp = 1;
104
105 s8tmp = hif_receive(u32Addr, (uint8 *)&strTlsSrvChunkRsp, sizeof(tstrTlsSrvChunkHdr), 0);
106 if(s8tmp == M2M_SUCCESS)
107 {
108 uint16 offset = strTlsSrvChunkRsp.u16Offset32;
109 uint16 chunk_size = strTlsSrvChunkRsp.u16Size32;
110 uint16 total_size = strTlsSrvChunkRsp.u16TotalSize32;
111 tenuTlsFlashStatus status = (tenuTlsFlashStatus)(strTlsSrvChunkRsp.u16Sig);
112
113 /* If first chunk, reset status. */
114 if(offset == 0)
115 genuStatus = TLS_FLASH_OK_NO_CHANGE;
116 /* Only send status to app when processing last chunk. */
117 if(offset + chunk_size != total_size)
118 bCallApp = 0;
119
120 switch(status)
121 {
122 case TLS_FLASH_OK:
123 // Good flash write. Update status if no errors yet.
124 if(genuStatus == TLS_FLASH_OK_NO_CHANGE)
125 genuStatus = status;
126 break;
127 case TLS_FLASH_OK_NO_CHANGE:
128 // No change, don't update status.
129 break;
130 case TLS_FLASH_ERR_CORRUPT:
131 // Corrupt. Always update status.
132 genuStatus = status;
133 break;
134 case TLS_FLASH_ERR_NO_CHANGE:
135 // Failed flash write. Update status if no more serious error.
136 if((genuStatus != TLS_FLASH_ERR_CORRUPT) && (genuStatus != TLS_FLASH_ERR_UNKNOWN))
137 genuStatus = status;
138 break;
139 default:
140 // Don't expect any other case. Ensure we don't mask a previous corrupt error.
141 if(genuStatus != TLS_FLASH_ERR_CORRUPT)
142 genuStatus = TLS_FLASH_ERR_UNKNOWN;
143 break;
144 }
145 }
146 if(bCallApp && gpfAppSSLCb)
147 gpfAppSSLCb(M2M_SSL_RESP_WRITE_OWN_CERTS, &genuStatus);
148 }
149 break;
150 }
151 if(s8tmp != M2M_SUCCESS)
152 {
153 M2M_ERR("Error receiving SSL from the HIF\n");
154 }
155}

207NMI_API sint8 m2m_ssl_send_certs_to_winc(uint8 *pu8Buffer, uint32 u32BufferSz)
208{
209 sint8 s8Ret = M2M_SUCCESS;
210#define TXLIMIT (256 * 6)
211
212 if(u32BufferSz <= TXLIMIT)
213 {
214 // set chunk header for one chunk
215#pragma GCC diagnostic push
216#pragma GCC diagnostic ignored "-Wcast-align"
217 tstrTlsSrvChunkHdr *pchkhdr = (tstrTlsSrvChunkHdr *)pu8Buffer;
218#pragma GCC diagnostic pop
219 pchkhdr->u16Sig = TLS_CERTS_CHUNKED_SIG_VALUE;
220 pchkhdr->u16TotalSize32 = (u32BufferSz + 3) >> 2;
221 pchkhdr->u16Offset32 = 0;
222 pchkhdr->u16Size32 = (u32BufferSz + 3) >> 2;
223 s8Ret = hif_send(M2M_REQ_GROUP_SSL, (M2M_SSL_REQ_WRITE_OWN_CERTS | M2M_REQ_DATA_PKT), NULL, 0, pu8Buffer, u32BufferSz, 0);
224 M2M_INFO("Transferred %lu bytes of cert data NON-CHUNKED\n", u32BufferSz);
225 }
226 else
227 {
228 // chunk it
229 // We are sneaking in a header - tstrTlsSrvChunkHdr
230#define CHUNKHDRSZ (sizeof(tstrTlsSrvChunkHdr))
231#define CHUNKSZ (TXLIMIT - 256) // divisible by 4
232 uint8 saveblob[CHUNKHDRSZ];
233 uint32 ofs = 0;
234 uint32 thischunksz = 0;
235
236 // first is special - over writing our header
237 m2m_memcpy(saveblob, &pu8Buffer[ofs], CHUNKHDRSZ);
238 thischunksz = min(CHUNKSZ, u32BufferSz-ofs); // no need to round up to quad words this time
239#pragma GCC diagnostic push
240#pragma GCC diagnostic ignored "-Wcast-align"
241 tstrTlsSrvChunkHdr *pchkhdr = (tstrTlsSrvChunkHdr *)&pu8Buffer[ofs];
242#pragma GCC diagnostic pop
243 pchkhdr->u16Sig = TLS_CERTS_CHUNKED_SIG_VALUE;
244 pchkhdr->u16TotalSize32 = ((u32BufferSz + 3) >> 2);
245 pchkhdr->u16Offset32 = ((ofs + 3) >> 2);
246 pchkhdr->u16Size32 = ((thischunksz + 3) >> 2);
247 s8Ret = hif_send(M2M_REQ_GROUP_SSL, (M2M_SSL_REQ_WRITE_OWN_CERTS | M2M_REQ_DATA_PKT), NULL, 0, &pu8Buffer[ofs], thischunksz, 0);
248 M2M_INFO("Transferred %u bytes of cert data CHUNKED to offset %u total %u\n", thischunksz, ofs, u32BufferSz);
249 m2m_memcpy(&pu8Buffer[ofs], saveblob, CHUNKHDRSZ);
250 ofs += thischunksz;
251
252 while(ofs < u32BufferSz)
253 {
254 // Subsequent chunks write header before and send a little more
255 m2m_memcpy(saveblob, &pu8Buffer[ofs-CHUNKHDRSZ], CHUNKHDRSZ);
256 thischunksz = min(CHUNKSZ, u32BufferSz-ofs);
257 thischunksz = (thischunksz + 3) & 0xFFFFFFFC; // needs to round up to quad word length
258#pragma GCC diagnostic push
259#pragma GCC diagnostic ignored "-Wcast-align"
260 pchkhdr = (tstrTlsSrvChunkHdr *)&pu8Buffer[ofs - CHUNKHDRSZ];
261#pragma GCC diagnostic pop
262 pchkhdr->u16Sig = TLS_CERTS_CHUNKED_SIG_VALUE;
263 pchkhdr->u16TotalSize32 = ((u32BufferSz + 3) >> 2);
264 pchkhdr->u16Offset32 = ((ofs + 3) >> 2);
265 pchkhdr->u16Size32 = ((thischunksz + 3) >> 2);
266 s8Ret = hif_send(M2M_REQ_GROUP_SSL, (M2M_SSL_REQ_WRITE_OWN_CERTS | M2M_REQ_DATA_PKT), NULL, 0, &pu8Buffer[ofs - CHUNKHDRSZ], thischunksz + CHUNKHDRSZ, 0);
267 M2M_INFO("Transferred %lu bytes of cert data CHUNKED to offset %u total %u\n", thischunksz, ofs, u32BufferSz);
268 m2m_memcpy(&pu8Buffer[ofs - CHUNKHDRSZ], saveblob, CHUNKHDRSZ);
269 ofs += thischunksz;
270 }
271 }
272
273 return s8Ret;
274}
