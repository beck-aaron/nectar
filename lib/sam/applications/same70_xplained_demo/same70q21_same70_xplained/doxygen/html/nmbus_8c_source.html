197/*
198* @fn nm_read_block
199* @brief Read block of data
200* @param [in] u32Addr
201* Start address
202* @param [out] puBuf
203* Pointer to a buffer used to return the read data
204* @param [in] u32Sz
205* Number of bytes to read. The buffer size must be >= u32Sz
206* @return M2M_SUCCESS in case of success and M2M_ERR_BUS_FAIL in case of failure
207* @author M. Abdelmawla
208* @date 11 July 2012
209* @version 1.0
210*/
211sint8 nm_read_block(uint32 u32Addr, uint8 *puBuf, uint32 u32Sz)
212{
213 uint16 u16MaxTrxSz = egstrNmBusCapabilities.u16MaxTrxSz - MAX_TRX_CFG_SZ;
214 uint32 off = 0;
215 sint8 s8Ret = M2M_SUCCESS;
216
217 for(;;)
218 {
219 if(u32Sz <= u16MaxTrxSz)
220 {
221 s8Ret += p_nm_read_block(u32Addr, &puBuf[off], (uint16)u32Sz);
222 break;
223 }
224 else
225 {
226 s8Ret += p_nm_read_block(u32Addr, &puBuf[off], u16MaxTrxSz);
227 if(M2M_SUCCESS != s8Ret) break;
228 u32Sz -= u16MaxTrxSz;
229 off += u16MaxTrxSz;
230 u32Addr += u16MaxTrxSz;
231 }
232 }
233
234 return s8Ret;
235}
