274sint8 chip_sleep(void)
275{
276 uint32 reg;
277 sint8 ret = M2M_SUCCESS;
278
279 while(1)
280 {
281 ret = nm_read_reg_with_ret(CORT_HOST_COMM,&reg);
282 if(ret != M2M_SUCCESS) goto ERR1;
283 if((reg & NBIT0) == 0) break;
284 }
285
286 /* Clear bit 1 */

288 if(ret != M2M_SUCCESS)goto ERR1;

293 if(ret != M2M_SUCCESS)goto ERR1;
294 }

308sint8 chip_wake(void)
309{
310 sint8 ret = M2M_SUCCESS;
311 uint32 reg = 0, clk_status_reg = 0,trials = 0;
312
313 ret = nm_read_reg_with_ret(HOST_CORT_COMM, &reg);
314 if(ret != M2M_SUCCESS)goto _WAKE_EXIT;
315
316 if(!(reg & NBIT0))
317 {
318 /*USE bit 0 to indicate host wakeup*/
319 ret = nm_write_reg(HOST_CORT_COMM, reg|NBIT0);
320 if(ret != M2M_SUCCESS)goto _WAKE_EXIT;
321 }
322
323 ret = nm_read_reg_with_ret(WAKE_CLK_REG, &reg);
324 if(ret != M2M_SUCCESS)goto _WAKE_EXIT;
325 /* Set bit 1 */
326 if(!(reg & NBIT1))
327 {
328 ret = nm_write_reg(WAKE_CLK_REG, reg | NBIT1);
329 if(ret != M2M_SUCCESS) goto _WAKE_EXIT;
330 }
331
332 do
333 {
334 ret = nm_read_reg_with_ret(CLOCKS_EN_REG, &clk_status_reg);
335 if(ret != M2M_SUCCESS) {
336 M2M_ERR("Bus error (5).%d %lx\n",ret,clk_status_reg);
337 goto _WAKE_EXIT;
338 }
339 if(clk_status_reg & NBIT2) {
340 break;
341 }
342 nm_bsp_sleep(2);
343 trials++;
344 if(trials > WAKUP_TRAILS_TIMEOUT)
345 {
346 M2M_ERR("Failed to wake up the chip\n");
347 ret = M2M_ERR_TIME_OUT;
348 goto _WAKE_EXIT;
349 }
350 }while(1);
351
352 /*workaround sometimes spi fail to read clock regs after reading/writing clockless registers*/
353 nm_bus_reset();
354
355_WAKE_EXIT:
356 return ret;
357}

396sint8 wait_for_bootrom(uint8 arg)
397{
398 sint8 ret = M2M_SUCCESS;
399 uint32 reg = 0, cnt = 0;
400 uint32 u32GpReg1 = 0;
401 uint32 u32DriverVerInfo = M2M_MAKE_VERSION_INFO(M2M_RELEASE_VERSION_MAJOR_NO,\
402 M2M_RELEASE_VERSION_MINOR_NO, M2M_RELEASE_VERSION_PATCH_NO,\
403 M2M_MIN_REQ_DRV_VERSION_MAJOR_NO, M2M_MIN_REQ_DRV_VERSION_MINOR_NO,\
404 M2M_MIN_REQ_DRV_VERSION_PATCH_NO);
405
406
407 reg = 0;
408 while(1) {
409 reg = nm_read_reg(0x1014); /* wait for efuse loading done */
410 if (reg & 0x80000000) {
411 break;
412 }
413 nm_bsp_sleep(1); /* TODO: Why bus error if this delay is not here. */
414 }
415 reg = nm_read_reg(M2M_WAIT_FOR_HOST_REG);
416 reg &= 0x1;
417
418 /* check if waiting for the host will be skipped or not */
419 if(reg == 0)
420 {
421 reg = 0;
422 while(reg != M2M_FINISH_BOOT_ROM)
423 {
424 nm_bsp_sleep(1);
425 reg = nm_read_reg(BOOTROM_REG);
426
427 if(++cnt > TIMEOUT)
428 {
429 M2M_DBG("failed to load firmware from flash.\n");
430 ret = M2M_ERR_INIT;
431 goto ERR2;
432 }
433 }
434 }
435
436 if(M2M_WIFI_MODE_ATE_HIGH == arg) {
437 nm_write_reg(NMI_REV_REG, M2M_ATE_FW_START_VALUE);
438 nm_write_reg(NMI_STATE_REG, NBIT20);
439 }else if(M2M_WIFI_MODE_ATE_LOW == arg) {
440 nm_write_reg(NMI_REV_REG, M2M_ATE_FW_START_VALUE);
441 nm_write_reg(NMI_STATE_REG, 0);
442 }else if(M2M_WIFI_MODE_ETHERNET == arg){
443 u32GpReg1 = rHAVE_ETHERNET_MODE_BIT;
444 nm_write_reg(NMI_STATE_REG, u32DriverVerInfo);
445 } else {
446 /*bypass this step*/
447 nm_write_reg(NMI_STATE_REG, u32DriverVerInfo);
448 }
449
450 if(REV(nmi_get_chipid()) >= REV_3A0){
451 chip_apply_conf(u32GpReg1 | rHAVE_USE_PMU_BIT);
452 } else {
453 chip_apply_conf(u32GpReg1);
454 }
455 M2M_INFO("DriverVerInfo: 0x%08lx\n",u32DriverVerInfo);
456
457 nm_write_reg(BOOTROM_REG,M2M_START_FIRMWARE);
458
459#ifdef __ROM_TEST__
460 rom_test();
461#endif /* __ROM_TEST__ */
462
463ERR2:
464 return ret;
465}
