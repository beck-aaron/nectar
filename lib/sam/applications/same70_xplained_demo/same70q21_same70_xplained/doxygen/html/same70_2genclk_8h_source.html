123static inline void genclk_config_set_source(struct genclk_config *p_cfg,
124 enum genclk_source e_src)
125{
126 p_cfg->ctrl &= (~PMC_PCK_CSS_Msk);
127
128 switch (e_src) {
129 case GENCLK_PCK_SRC_SLCK_RC:
130 case GENCLK_PCK_SRC_SLCK_XTAL:
131 case GENCLK_PCK_SRC_SLCK_BYPASS:
132 p_cfg->ctrl |= (PMC_PCK_CSS_SLOW_CLK);
133 break;
134
135 case GENCLK_PCK_SRC_MAINCK_4M_RC:
136 case GENCLK_PCK_SRC_MAINCK_8M_RC:
137 case GENCLK_PCK_SRC_MAINCK_12M_RC:
138 case GENCLK_PCK_SRC_MAINCK_XTAL:
139 case GENCLK_PCK_SRC_MAINCK_BYPASS:
140 p_cfg->ctrl |= (PMC_PCK_CSS_MAIN_CLK);
141 break;
142
143 case GENCLK_PCK_SRC_PLLACK:
144 p_cfg->ctrl |= (PMC_PCK_CSS_PLLA_CLK);
145 break;
146
147 case GENCLK_PCK_SRC_MCK:
148 p_cfg->ctrl |= (PMC_PCK_CSS_MCK);
149 break;
150
151 default:
152 break;
153 }

193 case GENCLK_PCK_SRC_SLCK_BYPASS:

195 osc_enable(OSC_SLCK_32K_BYPASS);

252static inline uint32_t genclk_get_frequency_hz(uint32_t ul_id)
253{
254 uint32_t pck_source,clock_config;
255 uint32_t pck_freq = 0;
256 pck_source = (PMC->PMC_PCK[ul_id] & PMC_PCK_CSS_Msk) >> PMC_PCK_CSS_Pos;
257 switch(pck_source)
258 {
259 case PMC_PCK_CSS_SLOW_CLK:
260 if (pmc_get_slck_config())
261 pck_freq = OSC_SLCK_32K_RC_HZ;
262 else
263 pck_freq = OSC_SLCK_32K_XTAL_HZ;
264 break;
265 case PMC_PCK_CSS_MAIN_CLK:
266 case PMC_PCK_CSS_PLLA_CLK:
267 clock_config = pmc_get_mainck_config();
268 if (clock_config & CKGR_MOR_MOSCSEL)
269 {
270 switch(clock_config & CKGR_MOR_MOSCRCF_Msk)
271 {
272 case CKGR_MOR_MOSCRCF_4_MHz:
273 pck_freq = OSC_MAINCK_4M_RC_HZ;
274 break;
275 case CKGR_MOR_MOSCRCF_8_MHz:
276 pck_freq = OSC_MAINCK_8M_RC_HZ;
277 break;
278 case CKGR_MOR_MOSCRCF_12_MHz:
279 pck_freq = OSC_MAINCK_12M_RC_HZ;
280 break;
281 }
282 }
283 else
284 {
285 pck_freq = OSC_MAINCK_XTAL_HZ;
286 }
287
288 if (pck_source == PMC_PCK_CSS_PLLA_CLK)
289 {
290 clock_config = pmc_get_pllack_config();
291 if ((clock_config & CKGR_PLLAR_DIVA_Msk) == 0)
292 return 0;
293 pck_freq *= ((((clock_config & CKGR_PLLAR_MULA_Msk)>>CKGR_PLLAR_MULA_Pos) + 1)/(clock_config & CKGR_PLLAR_DIVA_Msk));
294 }
295 break;
296 case PMC_PCK_CSS_UPLL_CLK:
297 pck_freq = pmc_get_upllckdiv_config() ? PLL_UPLL_HZ/2: PLL_UPLL_HZ;
298 break;
299 case PMC_PCK_CSS_MCK:
300 pck_freq = sysclk_get_main_hz();
301 break;
302 default:
303 break;
304 }
305 pck_freq /= ((PMC->PMC_PCK[ul_id] & PMC_PCK_PRES_Msk) >> PMC_PCK_PRES_Pos)+1;
306 return pck_freq;
307}
