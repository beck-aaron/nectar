158void sysclk_init(void)
159{
160 struct pll_config pllcfg;
161
162 /* Set flash wait state to max in case the below clock switching. */
163 system_init_flash(CHIP_FREQ_CPU_MAX/2);
164
165 /* Config system clock setting */
166 if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_SLCK_RC) {
167 osc_enable(OSC_SLCK_32K_RC);
168 osc_wait_ready(OSC_SLCK_32K_RC);
169 pmc_mck_set_division(CONFIG_SYSCLK_DIV);
170 pmc_switch_mck_to_sclk(CONFIG_SYSCLK_PRES);
171 }
172
173 else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_SLCK_XTAL) {
174 osc_enable(OSC_SLCK_32K_XTAL);

178 }
179
180 else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_SLCK_BYPASS) {
181 osc_enable(OSC_SLCK_32K_BYPASS);
182 osc_wait_ready(OSC_SLCK_32K_BYPASS);
183 pmc_mck_set_division(CONFIG_SYSCLK_DIV);
184 pmc_switch_mck_to_sclk(CONFIG_SYSCLK_PRES);
185 }
186
187 else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_MAINCK_4M_RC) {
188 /* Already running from SYSCLK_SRC_MAINCK_4M_RC */
189 }
190
191 else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_MAINCK_8M_RC) {
192 osc_enable(OSC_MAINCK_8M_RC);
193 osc_wait_ready(OSC_MAINCK_8M_RC);
194 pmc_mck_set_division(CONFIG_SYSCLK_DIV);
195 pmc_switch_mck_to_mainck(CONFIG_SYSCLK_PRES);
196 }
197
198 else if (CONFIG_SYSCLK_SOURCE == SYSCLK_SRC_MAINCK_12M_RC) {
199 osc_enable(OSC_MAINCK_12M_RC);
200 osc_wait_ready(OSC_MAINCK_12M_RC);
201 pmc_mck_set_division(CONFIG_SYSCLK_DIV);

216 pmc_switch_mck_to_mainck(CONFIG_SYSCLK_PRES);
217 }
218
219#ifdef CONFIG_PLL0_SOURCE

222 pll_config_defaults(&pllcfg, 0);
223 pll_enable(&pllcfg, 0);
224 pll_wait_for_lock(0);
225 pmc_mck_set_division(CONFIG_SYSCLK_DIV);
226 pmc_switch_mck_to_pllack(CONFIG_SYSCLK_PRES);
227 }

231 pll_enable_source(CONFIG_PLL1_SOURCE);
232 pll_config_defaults(&pllcfg, 1);
233 pll_enable(&pllcfg, 1);
234 pll_wait_for_lock(1);
235 pmc_mck_set_division(CONFIG_SYSCLK_DIV);
236 pmc_switch_mck_to_upllck(CONFIG_SYSCLK_PRES);
237 }
238 /* Update the SystemFrequency variable */
239 SystemCoreClockUpdate();
240
241 /* Set a flash wait state depending on the master clock frequency */
242 system_init_flash(sysclk_get_cpu_hz() / CONFIG_SYSCLK_DIV);
243
244#if (defined CONFIG_SYSCLK_DEFAULT_RETURNS_SLOW_OSC)
245 /* Signal that the internal frequencies are setup */
246 sysclk_initialized = 1;
247#endif
