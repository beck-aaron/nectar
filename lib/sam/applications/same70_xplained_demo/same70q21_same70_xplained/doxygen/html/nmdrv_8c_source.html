54sint8 nm_get_firmware_info(tstrM2mRev* M2mRev)
55{
56 uint16 curr_drv_ver, min_req_drv_ver,curr_firm_ver;
57 uint32 reg = 0;
58 sint8 ret = M2M_SUCCESS;
59
60 ret = nm_read_reg_with_ret(NMI_REV_REG, &reg);
61 //In case the Firmware running is ATE fw
62 if(M2M_ATE_FW_IS_UP_VALUE == reg)
63 {
64 //Read FW info again from the register specified for ATE
65 ret = nm_read_reg_with_ret(NMI_REV_REG_ATE, &reg);
66 }
67 M2mRev->u8DriverMajor = M2M_GET_DRV_MAJOR(reg);
68 M2mRev->u8DriverMinor = M2M_GET_DRV_MINOR(reg);
69 M2mRev->u8DriverPatch = M2M_GET_DRV_PATCH(reg);
70 M2mRev->u8FirmwareMajor = M2M_GET_FW_MAJOR(reg);
71 M2mRev->u8FirmwareMinor = M2M_GET_FW_MINOR(reg);
72 M2mRev->u8FirmwarePatch = M2M_GET_FW_PATCH(reg);
73 M2mRev->u32Chipid = nmi_get_chipid();
74 M2mRev->u16FirmwareSvnNum = 0;
75
76 curr_firm_ver = M2M_MAKE_VERSION(M2mRev->u8FirmwareMajor, M2mRev->u8FirmwareMinor,M2mRev->u8FirmwarePatch);
77 curr_drv_ver = M2M_MAKE_VERSION(M2M_RELEASE_VERSION_MAJOR_NO, M2M_RELEASE_VERSION_MINOR_NO, M2M_RELEASE_VERSION_PATCH_NO);
78 min_req_drv_ver = M2M_MAKE_VERSION(M2mRev->u8DriverMajor, M2mRev->u8DriverMinor,M2mRev->u8DriverPatch);
79 if(curr_drv_ver < min_req_drv_ver) {
80 /*The current driver version should be larger or equal
81 than the min driver that the current firmware support */
82 ret = M2M_ERR_FW_VER_MISMATCH;
83 }
84 if(curr_drv_ver > curr_firm_ver) {
85 /*The current driver should be equal or less than the firmware version*/
86 ret = M2M_ERR_FW_VER_MISMATCH;
87 }
88 return ret;
89}

306sint8 nm_drv_init_start(void * arg)
307{
308 sint8 ret = M2M_SUCCESS;
309 uint8 u8Mode = M2M_WIFI_MODE_NORMAL;
310
311 if(NULL != arg) {
312 u8Mode = *((uint8 *)arg);
313 if((u8Mode < M2M_WIFI_MODE_NORMAL)||(u8Mode >= M2M_WIFI_MODE_MAX)) {
314 u8Mode = M2M_WIFI_MODE_NORMAL;
315 }
316 }
317
318 ret = wait_for_bootrom(u8Mode);
319 if (M2M_SUCCESS != ret) {
320 goto ERR2;
321 }
322
323 ret = wait_for_firmware_start(u8Mode);
324 if (M2M_SUCCESS != ret) {
325 goto ERR2;
326 }
327
328 if((M2M_WIFI_MODE_ATE_HIGH == u8Mode)||(M2M_WIFI_MODE_ATE_LOW == u8Mode)) {
329 goto ERR1;
330 } else {
331 /*continue running*/
332 }
333
334 ret = enable_interrupts();
335 if (M2M_SUCCESS != ret) {
336 M2M_ERR("failed to enable interrupts..\n");
337 goto ERR2;
338 }
339
340 return ret;
341ERR2:
342 nm_bus_iface_deinit();
343#ifdef CONF_WINC_USE_SPI
344 nm_spi_deinit();
345#endif
346ERR1:
347 return ret;
348}

393 goto ERR1;
394 }
395

397 if (M2M_SUCCESS != ret) {
398 M2M_ERR("[nmi stop]: fail init bus\n");
399 goto ERR1;
400 }
401#ifdef CONF_WINC_USE_SPI
402 /* Must do this after global reset to set SPI data packet size. */
403 nm_spi_deinit();
404#endif
405
406ERR1:
