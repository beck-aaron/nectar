154static void translate_address(Efc **pp_efc, uint32_t ul_addr,
155 uint16_t *pus_page, uint16_t *pus_offset)
156{
157 Efc *p_efc;
158 uint16_t us_page;
159 uint16_t us_offset;
160
161#if (SAM3XA || SAM3U4)
162 if (ul_addr >= IFLASH1_ADDR) {
163 p_efc = EFC1;
164 us_page = (ul_addr - IFLASH1_ADDR) / IFLASH1_PAGE_SIZE;
165 us_offset = (ul_addr - IFLASH1_ADDR) % IFLASH1_PAGE_SIZE;
166 } else {
167 p_efc = EFC0;
168 us_page = (ul_addr - IFLASH0_ADDR) / IFLASH0_PAGE_SIZE;
169 us_offset = (ul_addr - IFLASH0_ADDR) % IFLASH0_PAGE_SIZE;
170 }
171#elif (SAM4SD16 || SAM4SD32 || SAM4C32 || SAM4CMP32 || SAM4CMS32)
172 uint32_t uc_gpnvm2;
173 uc_gpnvm2 = flash_is_gpnvm_set(2);
174 if (ul_addr >= IFLASH1_ADDR) {
175 if(uc_gpnvm2 == FLASH_RC_YES) {
176 p_efc = EFC0;
177 } else {
178 p_efc = EFC1;
179 }
180 us_page = (ul_addr - IFLASH1_ADDR) / IFLASH1_PAGE_SIZE;
181 us_offset = (ul_addr - IFLASH1_ADDR) % IFLASH1_PAGE_SIZE;
182 } else {
183 if(uc_gpnvm2 == FLASH_RC_YES) {
184 p_efc = EFC1;
185 } else {
186 p_efc = EFC0;
187 }
188 us_page = (ul_addr - IFLASH0_ADDR) / IFLASH0_PAGE_SIZE;
189 us_offset = (ul_addr - IFLASH0_ADDR) % IFLASH0_PAGE_SIZE;
190 }
191#elif (SAM3S8 || SAM3SD8)
192 p_efc = EFC;
193 us_page = (ul_addr - IFLASH0_ADDR) / IFLASH0_PAGE_SIZE;
194 us_offset = (ul_addr - IFLASH0_ADDR) % IFLASH0_PAGE_SIZE;
195#else
196 Assert(ul_addr >= IFLASH_ADDR);
197 Assert(ul_addr <= (IFLASH_ADDR + IFLASH_SIZE));
198
199 p_efc = EFC;
200 us_page = (ul_addr - IFLASH_ADDR) / IFLASH_PAGE_SIZE;
201 us_offset = (ul_addr - IFLASH_ADDR) % IFLASH_PAGE_SIZE;
202#endif
203
204 /* Store values */
205 if (pp_efc) {
206 *pp_efc = p_efc;
207 }
208
209 if (pus_page) {
210 *pus_page = us_page;
211 }
212
213 if (pus_offset) {
214 *pus_offset = us_offset;
215 }

226static void compute_address(Efc *p_efc, uint16_t us_page, uint16_t us_offset,
227 uint32_t *pul_addr)
228{
229 uint32_t ul_addr;
230
231/* Dual bank flash */
232#ifdef EFC1

234#if (SAM4SD16 || SAM4SD32 || SAM4C32 || SAM4CMP32 || SAM4CMS32)
235 uint32_t uc_gpnvm2;
236 uc_gpnvm2 = flash_is_gpnvm_set(2);
237 if (p_efc == EFC0) {
238 if(uc_gpnvm2 == FLASH_RC_YES) {
239 ul_addr = IFLASH1_ADDR + us_page * IFLASH_PAGE_SIZE + us_offset;
240 } else {
241 ul_addr = IFLASH0_ADDR + us_page * IFLASH_PAGE_SIZE + us_offset;
242 }
243 } else {
244 if(uc_gpnvm2 == FLASH_RC_YES) {
245 ul_addr = IFLASH0_ADDR + us_page * IFLASH_PAGE_SIZE + us_offset;
246 } else {
247 ul_addr = IFLASH1_ADDR + us_page * IFLASH_PAGE_SIZE + us_offset;

252 IFLASH0_ADDR + us_page * IFLASH_PAGE_SIZE + us_offset :

402uint32_t flash_get_descriptor(uint32_t ul_address,
403 uint32_t *pul_flash_descriptor, uint32_t ul_size)
404{
405 Efc *p_efc;
406 uint32_t ul_tmp;
407 uint32_t ul_cnt;
408
409 translate_address(&p_efc, ul_address, NULL, NULL);
410
411 /* Command fails */
412 if (FLASH_RC_OK != efc_perform_command(p_efc, EFC_FCMD_GETD, 0)) {
413 return 0;
414 } else {
415 /* Read until no result */
416 for (ul_cnt = 0;; ul_cnt++) {
417 ul_tmp = efc_get_result(p_efc);
418 if ((ul_size > ul_cnt) && (ul_tmp != 0)) {
419 *pul_flash_descriptor++ = ul_tmp;
420 } else {
421 break;
422 }
423 }
424 }
425
426 return ul_cnt;
427}

597uint32_t flash_write(uint32_t ul_address, const void *p_buffer,
598 uint32_t ul_size, uint32_t ul_erase_flag)
599{
600 Efc *p_efc;
601 uint32_t ul_fws_temp;
602 uint16_t us_page;
603 uint16_t us_offset;
604 uint32_t writeSize;
605 uint32_t ul_page_addr;
606 uint16_t us_padding;
607 uint32_t ul_error;
608 uint32_t ul_idx;
609 uint32_t *p_aligned_dest;
610 uint8_t *puc_page_buffer = (uint8_t *) gs_ul_page_buffer;
611
612 translate_address(&p_efc, ul_address, &us_page, &us_offset);
613
614#if SAM3S || SAM3N || SAM3XA || SAM3U
615 /* According to the errata, set the wait state value to 6. */
616 ul_fws_temp = efc_get_wait_state(p_efc);
617 efc_set_wait_state(p_efc, 6);
618#else
619 UNUSED(ul_fws_temp);
620#endif
621
622 /* Write all pages */
623 while (ul_size > 0) {
624 /* Copy data in temporary buffer to avoid alignment problems. */
625 writeSize = Min((uint32_t) IFLASH_PAGE_SIZE - us_offset,
626 ul_size);
627 compute_address(p_efc, us_page, 0, &ul_page_addr);
628 us_padding = IFLASH_PAGE_SIZE - us_offset - writeSize;
629
630 /* Pre-buffer data */
631 memcpy(puc_page_buffer, (void *)ul_page_addr, us_offset);
632
633 /* Buffer data */
634 memcpy(puc_page_buffer + us_offset, p_buffer, writeSize);
635
636 /* Post-buffer data */
637 memcpy(puc_page_buffer + us_offset + writeSize,
638 (void *)(ul_page_addr + us_offset + writeSize),
639 us_padding);
640
641 /* Write page.
642 * Writing 8-bit and 16-bit data is not allowed and may lead to
643 * unpredictable data corruption.
644 */
645 p_aligned_dest = (uint32_t *) ul_page_addr;
646 for (ul_idx = 0; ul_idx < (IFLASH_PAGE_SIZE / sizeof(uint32_t));
647 ++ul_idx) {
648 *p_aligned_dest++ = gs_ul_page_buffer[ul_idx];
649 }
650
651 if (ul_erase_flag) {
652 ul_error = efc_perform_command(p_efc, EFC_FCMD_EWP,
653 us_page);
654 } else {
655 ul_error = efc_perform_command(p_efc, EFC_FCMD_WP,
656 us_page);
657 }
658
659 if (ul_error) {
660 return ul_error;
661 }
662
663 /* Progression */
664 p_buffer = (void *)((uint32_t) p_buffer + writeSize);
665 ul_size -= writeSize;
666 us_page++;
667 us_offset = 0;
668 }
669
670#if SAM3S || SAM3N || SAM3XA || SAM3U
671 /* According to the errata, restore the wait state value. */
672 efc_set_wait_state(p_efc, ul_fws_temp);
673#endif
674
675 return FLASH_RC_OK;
676}

690uint32_t flash_lock(uint32_t ul_start, uint32_t ul_end,
691 uint32_t *pul_actual_start, uint32_t *pul_actual_end)
692{
693 Efc *p_efc;
694 uint32_t ul_actual_start, ul_actual_end;
695 uint16_t us_start_page, us_end_page;
696 uint32_t ul_error;
697 uint16_t us_num_pages_in_region =
698 IFLASH_LOCK_REGION_SIZE / IFLASH_PAGE_SIZE;
699
700 /* Compute actual lock range and store it */
701 compute_lock_range(ul_start, ul_end, &ul_actual_start, &ul_actual_end);
702
703 if (pul_actual_start != NULL) {
704 *pul_actual_start = ul_actual_start;
705 }
706
707 if (pul_actual_end != NULL) {
708 *pul_actual_end = ul_actual_end;
709 }
710
711 /* Compute page numbers */
712 translate_address(&p_efc, ul_actual_start, &us_start_page, 0);
713 translate_address(0, ul_actual_end, &us_end_page, 0);
714
715 /* Lock all pages */
716 while (us_start_page < us_end_page) {
717 ul_error = efc_perform_command(p_efc, EFC_FCMD_SLB, us_start_page);
718
719 if (ul_error) {
720 return ul_error;
721 }
722 us_start_page += us_num_pages_in_region;
723 }
724
725 return FLASH_RC_OK;
726}

783uint32_t flash_is_locked(uint32_t ul_start, uint32_t ul_end)
784{
785 Efc *p_efc;
786 uint16_t us_start_page, us_end_page;
787 uint8_t uc_start_region, uc_end_region;
788 uint16_t us_num_pages_in_region;
789 uint32_t ul_status;
790 uint32_t ul_error;
791 uint32_t ul_num_locked_regions = 0;
792 uint32_t ul_count = 0;
793 uint32_t ul_bit = 0;
794
795 Assert(ul_end >= ul_start);
796
797#ifdef EFC1
798 Assert(((ul_start >= IFLASH0_ADDR)
799 && (ul_end <= IFLASH0_ADDR + IFLASH0_SIZE))
800 || ((ul_start >= IFLASH1_ADDR)
801 && (ul_end <= IFLASH1_ADDR + IFLASH1_SIZE)));
802#else
803 Assert((ul_start >= IFLASH_ADDR)
804 && (ul_end <= IFLASH_ADDR + IFLASH_SIZE));
805#endif
806
807 /* Compute page numbers */
808 translate_address(&p_efc, ul_start, &us_start_page, 0);
809 translate_address(0, ul_end, &us_end_page, 0);
810
811 /* Compute region numbers */
812 us_num_pages_in_region = IFLASH_LOCK_REGION_SIZE / IFLASH_PAGE_SIZE;
813 uc_start_region = us_start_page / us_num_pages_in_region;
814 uc_end_region = us_end_page / us_num_pages_in_region;
815
816 /* Retrieve lock status */
817 ul_error = efc_perform_command(p_efc, EFC_FCMD_GLB, 0);
818 if (ul_error) {
819 return ul_error;
820 }
821 UNUSED(ul_error);
822
823 /* Skip unrequested regions (if necessary) */
824 ul_status = efc_get_result(p_efc);
825 while (!(ul_count <= uc_start_region &&
826 uc_start_region < (ul_count + 32))) {
827 ul_status = efc_get_result(p_efc);
828 ul_count += 32;
829 }
830
831 /* Check status of each involved region */
832 ul_bit = uc_start_region - ul_count;
833
834 /* Number of region to check (must be > 0) */
835 ul_count = uc_end_region - uc_start_region + 1;
836
837 while (ul_count > 0) {
838 if (ul_status & (1 << (ul_bit))) {
839 ul_num_locked_regions++;
840 }
841
842 ul_count -= 1;
843 ul_bit += 1;
844 if (ul_bit == 32) {
845 ul_status = efc_get_result(p_efc);
846 ul_bit = 0;
847 }
848 }
849
850 return ul_num_locked_regions;
851}
