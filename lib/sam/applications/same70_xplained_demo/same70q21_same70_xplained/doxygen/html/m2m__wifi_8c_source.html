794 gu321xTlsHsFlags &= ~WIFI_1X_TLS_HS_FLAGS_REQUIRE_TIME;

801 gu321xTlsHsFlags |= WIFI_1X_TLS_HS_FLAGS_PEER_CERTTIMECHECK;
802 gu321xTlsHsFlags &= ~WIFI_1X_TLS_HS_FLAGS_REQUIRE_TIME;
803 break;

805 return M2M_ERR_INVALID_ARG;
806 }
807 break;
808 case WIFI_1X_SESSION_CACHING:

810 return M2M_ERR_INVALID_ARG;
811 switch(*(int*)pOptionValue)
812 {
813 case 1:

815 break;
816 case 0:
817 gu321xTlsHsFlags &= ~WIFI_1X_TLS_HS_FLAGS_SESSION_CACHING;
818 break;
819 default:
820 return M2M_ERR_INVALID_ARG;
821 }
822 break;
823 case WIFI_1X_SPECIFIC_ROOTCERT:
824 switch(OptionLen)

826 case 20:

1074sint8 m2m_wifi_connect_sc(char *pcSsid, uint8 u8SsidLen, uint8 u8SecType, void *pvAuthInfo, uint16 u16Ch, uint8 u8NoSaveCred)

1078 tenuCredStoreOption enuCredStoreOption = u8NoSaveCred ? WIFI_CRED_DONTSAVE : WIFI_CRED_SAVE_ENCRYPTED;
1079
1080 /* This API does not support SSIDs which contain '\0'. If there is a '\0' character within the
1081 * first u8SsidLen characters, then assume that the input u8SsidLen was incorrect - set length
1082 * to strlen(pcSsid) and continue. This is to avoid a change from the behaviour of previously
1083 * released drivers. */
1084 if (u8SsidLen < M2M_MAX_SSID_LEN)
1085 {
1086 while (u8SsidLen--)
1087 if (strNetworkId.pu8Ssid[u8SsidLen] == 0){
1088 strNetworkId.u8SsidLen = u8SsidLen;}
1089 }
1090
1091 switch ((tenuM2mSecType)u8SecType)
1092 {
1093 case M2M_WIFI_SEC_OPEN:
1094 s8Ret = m2m_wifi_connect_open(enuCredStoreOption, &strNetworkId);
1095 break;
1096 case M2M_WIFI_SEC_WPA_PSK:
1097 if (pvAuthInfo != NULL)
1098 {
1099 tstrAuthPsk strAuthPsk = {NULL, NULL, 0};
1100 uint16 len = m2m_strlen((uint8*)pvAuthInfo);
1101
1102 if (len == M2M_MAX_PSK_LEN-1)
1103 {
1104 strAuthPsk.pu8Psk = (uint8*)pvAuthInfo;
1105 }
1106 else
1107 {
1108 strAuthPsk.pu8Passphrase = (uint8*)pvAuthInfo;
