131/* Convert hexchar to value 0-15 */
132static uint8 hexchar_2_val(uint8 ch)
133{
134 /* ch -= '0' */
135 ch -= 0x30;
136 if(ch <= 9)
137 return ch;
138 /* OR with 0x20 to convert upper case to lower case. */
139 ch |= 0x20;
140 /* ch -= ('a'-'0') */
141 ch -= 0x31;
142 if(ch <= 5)
143 return ch + 10;
144 return 0xFF;
145}
