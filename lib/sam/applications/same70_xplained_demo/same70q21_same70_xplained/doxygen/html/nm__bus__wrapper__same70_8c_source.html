118sint8 nm_spi_rw(uint8 *pu8Mosi, uint8 *pu8Miso, uint16 u16Sz)
119{
120 uint8 u8Dummy = 0;
121 uint8 u8SkipMosi = 0, u8SkipMiso = 0;
122 uint16_t txd_data = 0;
123 uint16_t rxd_data = 0;
124 uint8_t uc_pcs;
125
126 if (!pu8Mosi) {
127 pu8Mosi = &u8Dummy;
128 u8SkipMosi = 1;
129 }
130 else if(!pu8Miso) {
131 pu8Miso = &u8Dummy;
132 u8SkipMiso = 1;
133 }
134 else {
135 return M2M_ERR_BUS_FAIL;
136 }
137
138 SPI_ASSERT_CS();
139 while (u16Sz) {
140 txd_data = *pu8Mosi;
141 spi_write(CONF_WINC_SPI, txd_data, 0, 0);
142
143 /* Read SPI master data register. */
144 spi_read(CONF_WINC_SPI, &rxd_data, &uc_pcs);
145 *pu8Miso = rxd_data;
146
147 u16Sz--;
148 if (!u8SkipMiso)
149 pu8Miso++;
150 if (!u8SkipMosi)
151 pu8Mosi++;
152 }
153 SPI_DEASSERT_CS();
154
155 return M2M_SUCCESS;
156}

222/*
223 * @fn nm_bus_ioctl
224 * @brief send/receive from the bus
225 * @param[IN] u8Cmd
226 * IOCTL command for the operation
227 * @param[IN] pvParameter
228 * Arbitrary parameter depenging on IOCTL
229 * @return M2M_SUCCESS in case of success and M2M_ERR_BUS_FAIL in case of failure
230 * @note For SPI only, it's important to be able to send/receive at the same time
231 */
232sint8 nm_bus_ioctl(uint8 u8Cmd, void* pvParameter)
233{
234 sint8 s8Ret = 0;
235 switch(u8Cmd)
236 {
237#ifdef CONF_WINC_USE_I2C
238 case NM_BUS_IOCTL_R: {
239 tstrNmI2cDefault *pstrParam = (tstrNmI2cDefault *)pvParameter;
240 s8Ret = nm_i2c_read(pstrParam->pu8Buf, pstrParam->u16Sz);
241 }
242 break;
243 case NM_BUS_IOCTL_W: {
244 tstrNmI2cDefault *pstrParam = (tstrNmI2cDefault *)pvParameter;
245 s8Ret = nm_i2c_write(pstrParam->pu8Buf, pstrParam->u16Sz);
246 }
247 break;
248 case NM_BUS_IOCTL_W_SPECIAL: {
249 tstrNmI2cSpecial *pstrParam = (tstrNmI2cSpecial *)pvParameter;
250 s8Ret = nm_i2c_write_special(pstrParam->pu8Buf1, pstrParam->u16Sz1, pstrParam->pu8Buf2, pstrParam->u16Sz2);
251 }
252 break;
253#elif CONF_WINC_USE_SPI
254 case NM_BUS_IOCTL_RW: {
255 tstrNmSpiRw *pstrParam = (tstrNmSpiRw *)pvParameter;
256 s8Ret = nm_spi_rw(pstrParam->pu8InBuf, pstrParam->pu8OutBuf, pstrParam->u16Sz);
257 }
258 break;
259#endif
260 default:
261 s8Ret = -1;
262 M2M_ERR("Invalid IOCTL command!\n");
263 break;
264 }
265
266 return s8Ret;
267}
