63{
64 sint8 s8Ret = hif_chip_wake();
65 if(s8Ret != M2M_SUCCESS) goto _EXIT;
66
67 if(u8GpioNum >= M2M_PERIPH_GPIO_MAX) goto _EXIT1;
68
69 if(op == GPIO_OP_DIR) {
70 s8Ret = set_gpio_dir(u8GpioNum, u8InVal);
71 } else if(op == GPIO_OP_SET) {
72 s8Ret = set_gpio_val(u8GpioNum, u8InVal);
73 } else if(op == GPIO_OP_GET) {
74 s8Ret = get_gpio_val(u8GpioNum, pu8OutVal);
